# bookstore

Backend:
    - Installed Packages:
        - Cors
        - Express
        - Mongoose
        - Nodemon

Frontend:
    - Vite
        - npm create vite@latest
    - Tailwinds
        - Framework Guides choose Vite
        - npm install -D tailwindcss postcss autoprefixer
        - npx tailwindcss init
        - Insert ["./index.html",
        "./src/**/*.{js,ts,jsx,tsx}",] into content array of tailwind.config.js
    - React Router Dom
    - Axios
    - React-icons
